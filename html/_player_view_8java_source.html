94 private void renderHandCardsView() {

97 var handCards = appService.getHandCards(player.getId()).collect(Collectors.toList());
98
99 Point originPoint = getFirstCardPoint(handCards.size());
100 int offset = calculateOffset(handCardsView.getWidth(), handCards.size());
101
102 int i = 0;
103 for (var card : handCards) {
104 var cardView = new CardView(card, this::playCard);
105
106 cardView.setBounds(originPoint.x, originPoint.y,
107 cardView.getDimension().width, cardView.getDimension().height);
108 handCardsView.add(cardView, i++);
109 handCardsView.moveToFront(cardView);
110
111 originPoint.x += offset;
112 }
113
114 handCardsView.revalidate();

123 private Point getFirstCardPoint(int totalCards) {
