146 private void startDiscardPile() {
147 var card = drawPile.drawCard();
148 switch (card.getType()) {
149 case NUMBER, WILD_COLOR -> discard(card);
150 case SKIP -> {
151 discard(card);
152 players.next();
153 }
154 case REVERSE -> {
155 discard(card);
156 reverse();
157 }
158 case DRAW_TWO -> {
159 discard(card);
160 drawTwoCards(players.getCurrentPlayer());
161 players.next();
162 }
163 case WILD_DRAW_FOUR -> {
164 recreateDrawPile(card);
165 startDiscardPile();
166 }
167 default -> throw new UnsupportedOperationException("Unsupported card type " + card.getType());
168 }
